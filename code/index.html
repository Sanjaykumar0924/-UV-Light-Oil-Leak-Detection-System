import cv2 import numpy as np cap = cv2.VideoCapture(0) while True: ret, frame = cap.read() if not ret: break hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV) lower_fluorescent = np.array([35, 50, 50]) upper_fluorescent = np.array([90, 255, 255]) mask = cv2.inRange(hsv, lower_fluorescent, upper_fluorescent) kernel = np.ones((5, 5), np.uint8) mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel) mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel) contours, _ = cv2.findContours( mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE ) leak_detected = False for contour in contours: area = cv2.contourArea(contour) if area > 300: leak_detected = True x, y, w, h = cv2.boundingRect(contour) cv2.rectangle( frame, (x, y), (x + w, y + h), (0, 255, 0), 2 ) cv2.putText( frame, "Oil Leak Detected", (x, y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2 ) if leak_detected: cv2.putText( frame, "Leak Found", (20, 40), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 255), 3 ) cv2.imshow("UV Oil Leak Detection", frame) cv2.imshow("Fluorescent Mask", mask) if cv2.waitKey(1) & 0xFF == ord('q'): break cap.release() cv2.destroyAllWindows()
